def heapify_down(heap, n, i):
    smallest = i
    left = 2 * i + 1
    right = 2 * i + 2

    if left < n and heap[left] < heap[smallest]:
        smallest = left
    if right < n and heap[right] < heap[smallest]:
        smallest = right

    if smallest != i:
        temp = heap[i]
        heap[i] = heap[smallest]
        heap[smallest] = temp
        heapify_down(heap, n, smallest)


def heapify_up(heap, i):
    while i > 0:
        parent = (i - 1) // 2


        if heap[parent] > heap[i]:
            temp = heap[i]
            heap[i] = heap[parent]
            heap[parent] = temp
            i = parent
        else:
            break


def build_min_heap(heap, n):
    i = n // 2 - 1
    while i >= 0:
        heapify_down(heap, n, i)
        i -= 1



def distribute_tasks(tasks, m):
    heap = [0] * m
    size = m
    build_min_heap(heap, size)

    
    for t in tasks:
       heap[0] = heap[0] + t
       heapify_down(heap, size, 0)

    return heap

class MaxHeap:
    def __init__(self, capacity):
        self.capacity = capacity
        self.size = 0


        self.data = [0] * capacity  
    def insert(self, value):
        if self.size == self.capacity:
            return  # Heap full
        self.data[self.size] = value
        self.size += 1
        self._heapify_up(self.size - 1)

    def extract_max(self):
        if self.size == 0:
            return None


        maximum = self.data[0]
        self.data[0] = self.data[self.size - 1]
        self.size -= 1
        self._heapify_down(0)
        return maximum



    def _heapify_up(self, index):
        while index > 0:
            parent = (index - 1) // 2
            if self.data[index] > self.data[parent]:
                # swap
                temp = self.data[index]
                self.data[index] = self.data[parent]
                self.data[parent] = temp
                index = parent
            else:
                break

    def _heapify_down(self, index):
        while True:
            left = 2 * index + 1
            right = 2 * index + 2
            largest = index

            if left < self.size and self.data[left] > self.data[largest]:
                largest = left


            if right < self.size and self.data[right] > self.data[largest]:
                largest = right

            if largest != index:
                temp = self.data[index]


                self.data[index] = self.data[largest]
                self.data[largest] = temp
                index = largest
            else:
                break


def find_k_largest(nums, k):
    n = len(nums)
    heap = MaxHeap(n)

   
    i = 0
    while i < n:
        heap.insert(nums[i])
        i += 1

    
    result = [0] * k
    i = 0
    while i < k:
        result[i] = heap.extract_max()
        i += 1

    return result
