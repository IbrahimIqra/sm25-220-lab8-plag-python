class MinHeap:
    def __init__(self, cap):


        self.__heap_arr =[0] * cap
        self.__size = 0


    def insert(self, val):


        if self.__size >= len(self.__heap_arr):
            print("Heap is full!")
            return




        self.__heap_arr[self.__size] =val
        self.__size +=1




        self.__swim(self.__size - 1)


    def __swim(self, idx):


        p_idx = (idx - 1) // 2




        if idx> 0 and self.__heap_arr[idx] < self.__heap_arr[p_idx]:
          self.__heap_arr[idx],self.__heap_arr[p_idx] = self.__heap_arr[p_idx],self.__heap_arr[idx]


          self.__swim(p_idx)


    def extractMin(self):


        if self.__size <= 0:
          print("Heap is empty!")
          return None




        min =self.__heap_arr[0]


        self.__heap_arr[0] = self.__heap_arr[self.__size - 1]
        self.__size -= 1


        if self.__size > 0:
            self.__sink(0)


        return min


    def __sink(self, idx):
        smallest = idx
        left = 2*idx + 1
        right = 2 *idx+ 2






        if left <self.__size and self.__heap_arr[left]<self.__heap_arr[smallest]:
            smallest = left


        if right <self.__size and self.__heap_arr[right] < self.__heap_arr[smallest]:
            smallest = right


        if smallest != idx:
            self.__heap_arr[idx], self.__heap_arr[smallest] = self.__heap_arr[smallest], self.__heap_arr[idx]
            self.__sink(smallest)


    def get_array(self):


        return self.__heap_arr[:self.__size]


    def get_size(self):


        return self.__size




def distributeTasks(tasks, m):


    mloads = MinHeap(m)




    for i in range(m):
      mloads.insert(0)




    for i in tasks:


        min_load =mloads.extractMin()

        mloads.insert(updated_load)




    return  mloads.get_array()

def findTopKElements(nums, k):


    heap = MaxHeap(len(nums))


    for i in nums:
        heap.insert(i)




    res = []
    for i in range(k):
        res.append(heap.extractMax())


    return res
