class MinHeap:
    def __init__(self, capacity):
        self.__capacity = capacity
        self.__heap = [None] * (capacity + 1)
        self.__size = 0


    def insert(self, data):
        if self.__size >= self.__capacity:
            return False
        self.__size += 1
        self.__heap[self.__size] = data
        self.__swim(self.__size)
        return True


    def __swim(self, index):
        while index > 1 and self.__heap[index] < self.__heap[index // 2]:
            self.__heap[index], self.__heap[index//2] = self.__heap[index//2], self.__heap[index]


            index //= 2


    def extractMin(self):
        if self.__size == 0:
            return False
        min_val = self.__heap[1]
        self.__heap[1] = self.__heap[self.__size]
        self.__size -= 1
        self.__sink()
        return min_val


    def __sink(self, i=1):
        left_child = 2 * i
        right_child = 2 * i + 1
        smallest = i
        if left_child <= self.__size and self.__heap[left_child] < self.__heap[smallest]:
            smallest = left_child
        if right_child <= self.__size and self.__heap[right_child] < self.__heap[smallest]:
            smallest = right_child
        if smallest != i:
            self.__heap[i], self.__heap[smallest] = self.__heap[smallest], self.__heap[i]
            self.__sink(smallest)


    def sort(self):
        a=self.__heap
        sorted_array = [None] * self.__size
        sorted_index = 0
        original_size = self.__size
        for _ in range(original_size):
            min_val = self.extractMin()
            if min_val :
                sorted_array[sorted_index] = min_val
                sorted_index += 1
        self.__heap=a
        return sorted_array

            return False

        self.__heap[1] = self.__heap[self.__size]
        self.__size -= 1
        self.__sink()
        return max


    def __sink(self, i=1):
        l_child = 2*i
        r_child = 2*i + 1
        largest = i
        if l_child <= self.__size and self.__heap[l_child] > self.__heap[largest]:
            largest = l_child
        if r_child <= self.__size and self.__heap[r_child] > self.__heap[largest]:
            largest = r_child

            self.__heap[i], self.__heap[largest] = self.__heap[largest], self.__heap[i]
            self.__sink(largest)

def schedule_tasks(m, tasks):
    min_heap = MinHeap(m)
    i=0
    while i<m:
        i+=1
        min_heap.insert(0)
    for el in tasks:
        ext = min_heap.extractMin()
        if ext is not None:
            min_heap.insert(ext + el)


    return min_heap.sort()


m = 4

result = schedule_tasks(m, tasks)

def find_top(nums, k):
    max_heap = MaxHeap(len(nums))


    for el in nums:
        max_heap.insert(el)
    arr = [0]*k
    for i in range(k):
        arr[i] = max_heap.extractMax()
    return arr
