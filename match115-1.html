class MinHeap:
  def __init__ (self,capacity):
    self.__capacity = capacity
    self.__size = 0
    self.__heap = [0]*(capacity)
  def insert(self,value):
    if self.__size>=self.__capacity:
      return
    self.__heap[self.__size]=value
    self.swim(self.__size)
    self.__size+=1


  def swim(self,index):
    while index>0 :
      parent= (index-1)//2
      if self.__heap[index]<self.__heap[parent]:
        self.__heap[index],self.__heap[parent]=self.__heap[parent],self.__heap[index]
        index=parent
      else:
        break
  def extractMin(self):
    if self.__size==0:
      return
    min=self.__heap[0]
    self.__size-=1
    self.__heap[0]=self.__heap[self.__size]
    self.sink(0)
    return min
  def sink(self,index):


    while True:
      left=2*index+1
      right=2*index+2
      smallest=index

      if left<self.__size and self.__heap[left]<self.__heap[smallest]:
        smallest=left
      if right<self.__size and self.__heap[right]<self.__heap[smallest]:
        smallest=right

      if smallest!=index:
      
        self.__heap[index],self.__heap[smallest]=self.__heap[smallest],self.__heap[index]
        index=smallest
      else:
        break
  def sort(self):
        original_size = self.__size
        for i in range(original_size - 1, 0, -1):
            self.__heap[0], self.__heap[i] = self.__heap[i], self.__heap[0]
            self.__size -= 1
            self.sink(0)
        self.__size = original_size 

    
  def printHeap(self):
        for i in range(self.__size):
            print(self.__heap[i], end=" ")
        print()

    
  def getHeap(self):
        return self.__heap[:self.__size]
  def reset(self):
        self.__heap = [0] * self.__capacity
        self.__size = self.__capacity


def distribute_tasks(tasks, m):
    heap = MinHeap(m)
    heap.reset()

    for task in tasks:
        min_load = heap.extractMin()
        updated_load = min_load + task
        heap.insert(updated_load)

    return heap.getHeap()
    

tasks = [2, 4, 7, 1, 6]
m = 4

result = distribute_tasks(tasks, m)
print("Final machine loads:", result)

class MaxHeap:
    def __init__(self, capacity):
        self.__capacity = capacity
        self.__heap = [0] * capacity
        self.__size = 0

    def insert(self, value):
        if self.__size >= self.__capacity:
            return
        self.__heap[self.__size] = value
        self.swim(self.__size)
        self.__size += 1



    def swim(self, index):
        while index > 0:
            parent = (index - 1) // 2
            if self.__heap[index] > self.__heap[parent]:
                self.__heap[index], self.__heap[parent] = self.__heap[parent], self.__heap[index]
                index = parent
            else:
                break

    def extractMax(self):
        if self.__size == 0:
            return None
        max_val = self.__heap[0]
        self.__size -= 1
        self.__heap[0] = self.__heap[self.__size]
        self.sink(0)
        return max_val

    def sink(self, index):


        while True:
            left = 2 * index + 1
            right = 2 * index + 2
            largest = index

            if left < self.__size and self.__heap[left] > self.__heap[largest]:
                largest = left
            if right < self.__size and self.__heap[right] > self.__heap[largest]:
                largest = right

            if largest != index:
                self.__heap[index], self.__heap[largest] = self.__heap[largest], self.__heap[index]
                index = largest
            else:
                break

    def getHeap(self):
        return self.__heap[:self.__size]
def top_k_largest(nums, k):
    heap = MaxHeap(len(nums))



    for num in nums:
        heap.insert(num)

    result = [0] * k
    for i in range(k):
        result[i] = heap.extractMax()

    return result
nums = [4, 10, 2, 8, 6, 7]

top_k = top_k_largest(nums, k)
