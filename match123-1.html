class MinHeap:
    def __init__(self, capacity):
        self.capacity = capacity
        self.size = 0
        self.heap = [0] * capacity

    def insert(self, value):


        if self.size >= self.capacity:
            return
        self.heap[self.size] = value
        self.size += 1
        self.swim(self.size - 1)

    def swim(self, index):
        while index > 0:
            parent = (index - 1) // 2
            if self.heap[index] < self.heap[parent]:
                self.heap[index], self.heap[parent] = self.heap[parent], self.heap[index]
                index = parent


            else:
                break

    def extract(self):
        if self.size == 0:
            return None
        min_val = self.heap[0]
        self.heap[0] = self.heap[self.size - 1]
        self.size -= 1
        self.sink(0)
        return min_val

    def sink(self, index):
        while 2 * index + 1 < self.size:
            low = 2 * index + 1
            if low + 1 < self.size and self.heap[low + 1] < self.heap[low]:
                low += 1
            if self.heap[index] <= self.heap[low]:
                break
            self.heap[index], self.heap[low] = self.heap[low], self.heap[index]
            index = low

    def print_heap(self):
        for i in range(self.size):
            print(self.heap[i], end=' ')
        print()

def task(task, k):
    n = len(task)
    heap = MinHeap(k)
    for i in range(k):
        heap.insert(0)

    for i in range(n):
        min_load = heap.extract()
        min_load += task[i]
        heap.insert(min_load)

    heap.print_heap()

tasks = [2, 4, 7, 1, 6]
m = 4
task(tasks, m)

def largest(nums, k):
    n = len(nums)
    heap = MaxHeap(n)

    for i in range(n):
        heap.insert(nums[i])

    for i in range(k):
        val = heap.extract()
        print(val, end=' ')
    print()
