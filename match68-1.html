class MinHeap:
    def __init__(self, cap):
        self.cap = cap
        self.n = 0
        self.arr = [0] * cap

    def push(self, val):
        if self.n == self.cap:
            return
        self.arr[self.n] = val
        self.n += 1
        self._up(self.n - 1)



    def _up(self, i):
        while i > 0:
            p = (i - 1) // 2
            if self.arr[i] < self.arr[p]:
                self.arr[i], self.arr[p] = self.arr[p], self.arr[i]
                i = p
            else:
                break

    def pop(self):
        if self.n == 0:
            return None
        root = self.arr[0]
        self.arr[0] = self.arr[self.n - 1]
        self.n -= 1
        self._down(0)
        return root

    def _down(self, i):
        while True:


            l = 2 * i + 1
            r = 2 * i + 2
            s = i
            if l < self.n and self.arr[l] < self.arr[s]:
                s = l
            if r < self.n and self.arr[r] < self.arr[s]:
                s = r
            if s != i:
                self.arr[i], self.arr[s] = self.arr[s], self.arr[i]
                i = s
            else:
                break

    def toArray(self):
        return self.arr[:self.n]


class MaxHeap:
    def __init__(self, cap):
        self.cap = cap
        self.n = 0
        self.arr = [0] * cap

    def push(self, val):
        if self.n == self.cap:
            return
        self.arr[self.n] = val
        self.n += 1
        self._up(self.n - 1)

    def _up(self, i):
        while i > 0:


            p = (i - 1) // 2
            if self.arr[i] > self.arr[p]:
                self.arr[i], self.arr[p] = self.arr[p], self.arr[i]
                i = p
            else:
                break

    def pop(self):
        if self.n == 0:
            return None
        root = self.arr[0]
        self.arr[0] = self.arr[self.n - 1]
        self.n -= 1
        self._down(0)
        return root

    def _down(self, i):
        while True:


            l = 2 * i + 1
            r = 2 * i + 2
            big = i
            if l < self.n and self.arr[l] > self.arr[big]:
                big = l
            if r < self.n and self.arr[r] > self.arr[big]:
                big = r
            if big != i:
                self.arr[i], self.arr[big] = self.arr[big], self.arr[i]
                i = big
            else:
                break

    def toArray(self):
        return self.arr[:self.n]


def assignTasks(tasks, m):
    heap = MinHeap(m)
    for _ in range(m):
        heap.push(0)
    for t in tasks:
        cur = heap.pop()
        heap.push(cur + t)
    return sorted(heap.toArray())


def topK(nums, k):
    heap = MaxHeap(len(nums))
    for x in nums:
        heap.push(x)
    out = [0] * k
    for i in range(k):
        out[i] = heap.pop()
    return out
