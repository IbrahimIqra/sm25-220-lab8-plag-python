class MinHeap:
    def __init__(self, capacity):
        self.capacity = capacity
        self.size = capacity
        self.heap = [0] * capacity



    def heapify_down(self, i):
        left = 2 * i + 1
        right = 2 * i + 2
        smallest = i

        if left < self.size and self.heap[left] < self.heap[smallest]:
            smallest = left
        if right < self.size and self.heap[right] < self.heap[smallest]:
            smallest = right



        if smallest != i:
            temp = self.heap[i]
            self.heap[i] = self.heap[smallest]
            self.heap[smallest] = temp
            self.heapify_down(smallest)

    def heapify_up(self, i):
        while i > 0:
            parent = (i - 1) // 2


            if self.heap[i] < self.heap[parent]:
                temp = self.heap[i]
                self.heap[i] = self.heap[parent]
                self.heap[parent] = temp
                i = parent
            else:
                break

    def extract_min(self):
        if self.size == 0:
            return -1
        root = self.heap[0]
        self.heap[0] = self.heap[self.size - 1]
        self.size -= 1
        self.heapify_down(0)
        return root

    def insert(self, value):
        if self.size == self.capacity:
            return
        self.heap[self.size] = value
        self.size += 1
        self.heapify_up(self.size - 1)

    def replace_min(self, value):
        self.heap[0] = value
        self.heapify_down(0)


def distribute_tasks(tasks, m):
    heap = MinHeap(m)

    # assign tasks
    for t in tasks:
        smallest = heap.heap[0]
        heap.replace_min(smallest + t)

    return heap.heap

class MaxHeap:
    def __init__(self, capacity):
        self.capacity = capacity
        self.size = 0
        self.heap = [0] * capacity



    def heapify_down(self, i):
        left = 2 * i + 1
        right = 2 * i + 2
        largest = i

        if left < self.size and self.heap[left] > self.heap[largest]:
            largest = left
        if right < self.size and self.heap[right] > self.heap[largest]:
            largest = right

        if largest != i:


            temp = self.heap[i]
            self.heap[i] = self.heap[largest]
            self.heap[largest] = temp
            self.heapify_down(largest)



    def heapify_up(self, i):
        while i > 0:
            parent = (i - 1) // 2
            if self.heap[i] > self.heap[parent]:
                temp = self.heap[i]
                self.heap[i] = self.heap[parent]
                self.heap[parent] = temp
                i = parent
            else:
                break

    def insert(self, value):
        if self.size == self.capacity:
            return
        self.heap[self.size] = value
        self.size += 1
        self.heapify_up(self.size - 1)

    def extract_max(self):
        if self.size <= 0:
            return -1
        root = self.heap[0]
        self.heap[0] = self.heap[self.size - 1]
        self.size -= 1
        self.heapify_down(0)
        return root


def top_k_largest(nums, k):
    n = len(nums)
    heap = MaxHeap(n)

    for i in range(n):
        heap.insert(nums[i])



    result = [0] * k
    for i in range(k):
        result[i] = heap.extract_max()

    return result
