class MinHeap:
    def __init__(self,capcity):
        self.arr= [0]*capcity
        self.n=0
        self.cap= capcity

    def insert(self,val):
        if self.n>= self.cap:
            print("heap full!!")


            return
        self.arr[self.n]= val
        self.swimup(self.n)
        self.n +=1

    def swimup(self,idx):
        while idx>0:
            parent =(idx-1)//2
            if self.arr[idx]<self.arr[parent]:


                tmp = self.arr[idx]
                self.arr[idx]= self.arr[parent]
                self.arr[parent] =tmp
                idx=parent
            else:
                break

    def extractMin(self):


        if self.n== 0:
            return None
        ans =self.arr[0]
        self.arr[0]= self.arr[self.n-1]
        self.n -=1
        self.sinkdown(0)
        return ans

    def sinkdown(self,i):
        while True:
            l=2*i+1
            r=2*i+2
            sm=i


            if l<self.n and self.arr[l]<self.arr[sm]:
                sm = l
            if r< self.n and self.arr[r]< self.arr[sm]:
                sm= r
            if sm!= i:
                t= self.arr[i]


                self.arr[i]= self.arr[sm]
                self.arr[sm] =t
                i =sm
            else:
                break

    def sort(self):
        a =[]
        while self.n> 0:
            a.append(self.extractMin())
        return a

class MaxHeap:
    def __init__(self,capcity):
        self.data =[0]*capcity
        self.sz=0
        self.cap= capcity

    def insert(self,v):
        if self.sz>= self.cap:
            print("heap full!!")
            return
        self.data[self.sz]= v
        self.swim(self.sz)
        self.sz +=1



    def swim(self,i):
        while i>0:
            p=(i-1)//2
            if self.data[i]> self.data[p]:
                tmp=self.data[i]
                self.data[i]= self.data[p]
                self.data[p]= tmp
                i=p
            else:
                break

    def extractMax(self):
        if self.sz==0:


            return None
        ans=self.data[0]
        self.data[0]= self.data[self.sz-1]
        self.sz -=1
        self.sink(0)
        return ans

    def sink(self,i):
        while True:


            l=2*i+1
            r=2*i+2
            big=i
            if l< self.sz and self.data[l]> self.data[big]:
                big=l
            if r< self.sz and self.data[r]> self.data[big]:
                big=r
            if big!= i:
                t=self.data[i]
                self.data[i]= self.data[big]
                self.data[big]=t
                i=big
            else:
                break

    def sort(self):
        out=[]
        while self.sz>0:
            out.append(self.extractMax())
        return out


#task-3
def distribute_tasks(tasks,m):
    heap= MinHeap(m)
    for i in range(m):
        heap.insert(0)
    for t in tasks:
        x= heap.extractMin()
        newv=x+t
        heap.insert(newv)
    res=[]
    for j in range(heap.n):
        res.append(heap.arr[j])
    return res

def find_top_k_largest(nums,k):
    heap= MaxHeap(len(nums))
    for num in nums:
        heap.insert(num)
    ans=[]
    for i in range(k):
        ans.append(heap.extractMax())
    return ans
