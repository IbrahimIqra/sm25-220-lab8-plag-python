class MinHeap:
  def __init__(self,capacity):
    self.capacity=capacity
    self.size=capacity
    self.heap=self.create_array(capacity)
    self.initialize_heap()
  def create_array(self,length):
    arr=[0]*length
    return arr
  def initialize_heap(self):
    i=0
    while i<self.capacity:
      self.heap[i]=0
      i+=1
  def extract_min(self):
    if self.size==0:
      return-1


    min_val=self.heap[0]
    self.heap[0]=self.heap[self.size-1]
    self.size-=1
    self.heapify_down(0)
    return min_val
  def insert(self,value):
    if self.size==self.capacity:
      return
    self.heap[self.size]=value
    self.size+=1
    self.heapify_up(self.size-1)
  def heapify_up(self,index):
    while index>0:
      parent=(index-1)//2


      if self.heap[parent]>self.heap[index]:
        temp=self.heap[parent]
        self.heap[parent]=self.heap[index]
        self.heap[index]=temp
        index=parent
      else:
        break
  def heapify_down(self,index):
    while True:
      left=2*index+1
      right=2*index+2
      smallest=index
      if left<self.size and self.heap[left]<self.heap[smallest]:
        smallest=left
      if right<self.size and self.heap[right]<self.heap[smallest]:
        smallest=right
      if smallest!=index:


        temp=self.heap[index]
        self.heap[index]=self.heap[smallest]
        self.heap[smallest]=temp
        index=smallest
      else:
        break
  def get_heap_array(self):


    result=self.create_array(self.size)
    i=0
    while i<self.size:
      result[i]=self.heap[i]
      i+=1
    return result
def sort_array(arr):
  n=len(arr)
  i=0
  while i<n-1:
    j=i+1
    while j<n:
      if arr[i]>arr[j]:
        temp=arr[i]
        arr[i]=arr[j]
        arr[j]=temp
      j+=1
    i+=1
  return arr
def task(tasks,m):
  heap=MinHeap(m)
  task_len=len(tasks)
  i=0
  while i<task_len:
    min_load=heap.extract_min()
    updated_load=min_load+tasks[i]
    heap.insert(updated_load)
    i+=1
  unsorted=heap.get_heap_array()
  return sort_array(unsorted)
