import numpy as np
class MinHeap:
    def __init__(self, cap):


        self.heap = np.array([None] * cap)
        self.size = 0


    def insert(self, val):
        if (self.size >= len(self.heap)):
            print("Heap is full!")
            return None




        self.heap[self.size] = val
        self.size += 1
        self.swim(self.size - 1)

    def swim(self, idx):
        while (idx > 0) and (self.heap[idx] < self.heap[(idx - 1) // 2]):

            self.heap[idx], self.heap[(idx - 1) // 2] = self.heap[(idx - 1) // 2], self.heap[idx]
            idx = (idx - 1) // 2


    def extractMin(self):
        if self.size == 0:
            print("Heap is empty!")
            return None




        min = self.heap[0]


        self.size -= 1
        self.heap[0] = self.heap[self.size]

        self.sink(0)

        return min


    def sink(self, idx1):
        while (2 * (idx1 + 1) < self.size):
            left = (2 * idx1 + 1)
            right = (2 * idx1 + 2)
            smallest = idx1



            if (left < self.size) and (self.heap[left] < self.heap[smallest]):
                smallest = left
            if (right < self.size) and (self.heap[right] < self.heap[smallest]):
                smallest = right


            if smallest == idx1:
                break




            self.heap[idx1], self.heap[smallest] = self.heap[smallest], self.heap[idx1]
            idx1 = smallest


    def sort(self):
        original = self.size


        for i in range(self.size - 1, 0, -1):

            self.heap[0], self.heap[i] = self.heap[i], self.heap[0]
            self.size -= 1
            self.sink(0)

        self.size = original

        self.heap = np.array([None] * cap)
        self.size = 0


    def insert(self, val):
        if (self.size >= len(self.heap)):
            print("Heap is full!")
            return




        self.heap[self.size] = val
        self.size += 1
        self.swim(self.size - 1)

    def swim(self, i):
        while (i > 0) and (self.heap[i] > self.heap[(i - 1) // 2]):

            self.heap[i], self.heap[(i - 1) // 2] = self.heap[(i - 1) // 2], self.heap[i]
            i = (i - 1) // 2


    def extractMax(self):
        if self.size == 0:
            print("Heap is empty!")
            return None




        max = self.heap[0]
        self.size -= 1
        self.heap[0] = self.heap[self.size]
        self.sink(0)

        return max


    def sink(self, i):
        while (2 * (i+ 1) < self.size):
            left = (2 * i + 1)
            right = (2 * i + 2)
            largest = i



            if (left < self.size) and (self.heap[left] > self.heap[largest]):
                largest = left
            if (right < self.size) and (self.heap[right] > self.heap[largest]):
                largest = right


            if largest == i:
                break



            self.heap[i], self.heap[largest] = self.heap[largest], self.heap[i]
            i = largest


    def sort(self):
        original = self.size


        for j in range(self.size - 1, 0, -1):

            self.heap[0], self.heap[j] = self.heap[j], self.heap[0]
            self.size -= 1

            self.sink(0)

        self.size = original

def topK(num, k):
    heap = MaxHeap(len(num) + 1)




    for j in num:
        heap.insert(j)


    result = [0] * k
    for i in range(k):
        result[i] = heap.extractMax()

    return result
