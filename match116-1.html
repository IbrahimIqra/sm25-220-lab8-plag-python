class MinHeap:
    def __init__(self, size):
        self.data = [0]*size   
        self.size = size

    def left(self, i): return 2*i + 1
    def right(self, i): return 2*i + 2
    def parent(self, i): return (i-1)//2



    def heapify_down(self, i):
        smallest = i
        l = self.left(i)
        r = self.right(i)

        if l < self.size and self.data[l] < self.data[smallest]:
            smallest = l
        if r < self.size and self.data[r] < self.data[smallest]:
            smallest = r

        if smallest != i:
            self.data[i], self.data[smallest] = self.data[smallest], self.data[i]
            self.heapify_down(smallest)

    def add_task(self, t):
        self.data[0] += t
        self.heapify_down(0)


def distribute_tasks(tasks, m):
    heap = MinHeap(m)
    for t in tasks:
        heap.add_task(t)
    return heap.data

class MaxHeap:
  def __init__ (self,load):
    self.__heap=[0]*load
    self.__size=0
    self.__load=load
  def insert(self,value):
    if self.__size==self.__load:
      return False
    self.__heap[self.__size]=value
    self.__size+=1
    self.__swim(self.__size-1)
    return True
  def __swim(self,idx):
    var=(idx-1)//2
    while idx>0 and self.__heap[idx] > self.__heap[var]:
      a=self.__heap[var]
      self.__heap[var]=self.__heap[idx]
      self.__heap[idx]=a
      idx=var
      var=(idx-1)//2

  def extractMax(self):
    if self.__size==0:
      return None


    max=self.__heap[0]
    self.__heap[0]=self.__heap[self.__size-1]
    self.__size-=1

    self.__sink(0)   
    return max

  def __sink(self, idx):
    while True:
      left=2 * idx + 1
      right=2 * idx + 2
      large=idx



      if left<self.__size and self.__heap[left]>self.__heap[large]:
        large=left
      if right<self.__size and self.__heap[right]>self.__heap[large]:
        large=right

      if large!=idx:
        a=self.__heap[idx]
        self.__heap[idx]=self.__heap[large]
        self.__heap[large]=a
        idx=large
      else:
        break
    
  def Heap(self):
    arr=[0]*self.__size
    for i in range(self.__size):
      arr[i]=self.__heap[i]
    return arr

def top_k_largest(nums, k):


  maxHeap=MaxHeap(len(nums))
  for j in nums:
    maxHeap.insert(j)
  result=[0]*k 
  for i in range(k):
    result[i]=maxHeap.extractMax()
  return result
