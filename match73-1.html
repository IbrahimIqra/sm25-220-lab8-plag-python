# TASK 3: Distribute Loads
def distribute_loads(tasks, m):
    h = MinHeap(m)
    i = 0
    while i < m:
        h.insert(0)
        i += 1
    n = len(tasks)
    i = 0
    while i < n:
        cur = h.extractMin()
        cur = cur + tasks[i]
        h.insert(cur)
        i += 1
    return h.to_array()

def top_k(nums, k):
    n = len(nums)
    h = MaxHeap(n)
    i = 0
    while i < n:
        h.insert(nums[i])
        i += 1
    out = [0] * k
    i = 0
    while i < k:
        out[i] = h.extractMax()
        i += 1
    return out
