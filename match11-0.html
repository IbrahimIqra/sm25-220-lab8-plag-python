class MaxHeap:
  def __init__ (self,load):
    self.__heap=[0]*load
    self.__size=0
    self.__load=load
  def insert(self,value):
    if self.__size==self.__load:
      return False
    self.__heap[self.__size]=value
    self.__size+=1
    self.__swim(self.__size-1)
    return True
  def __swim(self,idx):
    var=(idx-1)//2
    while idx>0 and self.__heap[idx] > self.__heap[var]:
      a=self.__heap[var]
      self.__heap[var]=self.__heap[idx]
      self.__heap[idx]=a
      idx=var
      var=(idx-1)//2

  def extractMax(self):
    if self.__size==0:
      return None
    max=self.__heap[0]
    self.__heap[0]=self.__heap[self.__size-1]
    self.__size-=1

    self.__sink(0)   
    return max

  def __sink(self, idx):
    while True:
      left=2 * idx + 1
      right=2 * idx + 2
      large=idx

      if left<self.__size and self.__heap[left]>self.__heap[large]:
        large=left
      if right<self.__size and self.__heap[right]>self.__heap[large]:
        large=right

      if large!=idx:
        a=self.__heap[idx]
        self.__heap[idx]=self.__heap[large]
        self.__heap[large]=a
        idx=large
      else:
        break
    
  def Heap(self):
    arr=[0]*self.__size
    for i in range(self.__size):
      arr[i]=self.__heap[i]
    return arr

def top_k_largest(nums, k):
  maxHeap=MaxHeap(len(nums))
  for j in nums:
    maxHeap.insert(j)
  result=[0]*k 
  for i in range(k):
    result[i]=maxHeap.extractMax()
  return result
