class Minheap:
    def __init__(self,cap):
        self.cap=cap
        self.size=0
        self.arr=[0]*cap


    def parent(self,k):
        return (k-1)//2
    def left(self,k):
        return 2*k+1
    def right(self,k):
        return 2*k+2
    def insert(self,value):
        if(self.size==self.cap):
            return
        self.arr[self.size]=value
        self.size=self.size+1
        self.swim(self.size-1)
    def swim(self,k):
        while(k>0 and self.arr[self.parent(k)]>self.arr[k]):
            self.arr[self.parent(k)],self.arr[k] = self.arr[k],self.arr[self.parent(k)]
            k=self.parent(k)
    def extractMin(self):


        if(self.size<=0):
            return None
        root=self.arr[0]
        self.arr[0]=self.arr[self.size-1]
        self.size-=1
        self.sink(0)
        return root
    def sink(self,k):
        SMALLEST=k


        LEFT=self.left(k)
        RIGHT=self.right(k)
        if(LEFT<self.size and self.arr[LEFT]<self.arr[SMALLEST]):
            SMALLEST=LEFT
        if(RIGHT<self.size and self.arr[RIGHT]<self.arr[SMALLEST]):
            SMALLEST=RIGHT
        if(SMALLEST!=k):
            self.arr[k],self.arr[SMALLEST]=self.arr[SMALLEST],self.arr[k]


            self.sink(SMALLEST)
def distributeTasks(task,m):


    heap=Minheap(m)
    for k in range(m):
        heap.insert(0)
    for e in task:
        minLoad=heap.extractMin()
        heap.insert(minLoad+e) 
    RESULT=[0]*m
    for k in range(m):
        RESULT[k]=heap.extractMin()
    return RESULT

m=4

def TOPK(nums, k):
    heap=Maxheap(len(nums))
    for numss in nums:
        heap.insert(numss)
    result=[0]*k
    for j in range(k):
        result[j]=heap.extractMax()
    return result
