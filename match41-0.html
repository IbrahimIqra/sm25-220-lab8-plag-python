import numpy as np
class MinHeap:
  def __init__(self,arrayCapacity):
    self._array=np.array([None]*arrayCapacity)
    self._HeapSize=0

  def swim(self,Child_idx):
    parent=Child_idx//2
    if Child_idx==1:
      return


    elif self._array[ parent]> self._array[Child_idx]:
      self._array[ parent], self._array[Child_idx]= self._array[Child_idx],self._array[ parent]
      return self.swim(parent)
    else:
      return


  def insert(self,a):
    self._HeapSize+=1
    if len( self._array)> self._HeapSize:
      self._array[self._HeapSize]=a
      self.swim(self._HeapSize)
    else:
      print("Insert Failed")

  def sink(self,Parent_idx=1):
    left_idx=Parent_idx*2
    right_idx=(Parent_idx*2)+1



    if left_idx<=self._HeapSize and  right_idx<=self._HeapSize:

      if self._array[left_idx]<self._array[Parent_idx] and self._array[right_idx]<self._array[Parent_idx]:

        if self._array[left_idx]>=self._array[right_idx]:
          self._array[ Parent_idx], self._array[right_idx]= self._array[right_idx],self._array[Parent_idx]


          return self.sink(right_idx)
        else:
          self._array[ Parent_idx], self._array[left_idx]= self._array[left_idx],self._array[Parent_idx]
          return self.sink(left_idx)



      elif self._array[right_idx]<self._array[Parent_idx]:
        self._array[ Parent_idx], self._array[right_idx]= self._array[right_idx],self._array[Parent_idx]
        return self.sink(right_idx)



      elif self._array[left_idx]<self._array[Parent_idx]:
        self._array[ Parent_idx], self._array[left_idx]= self._array[left_idx],self._array[Parent_idx]
        return self.sink(left_idx)





    elif left_idx<=self._HeapSize and right_idx>self._HeapSize:
      if self._array[left_idx]<self._array[Parent_idx]:
        self._array[ Parent_idx], self._array[left_idx]= self._array[left_idx],self._array[Parent_idx]
        return self.sink(left_idx)

    elif  right_idx>self._HeapSize and left_idx>self._HeapSize:
      return

  def extractMin(self):


    x=self._array[1]
    self._array[1], self._array[self._HeapSize]= self._array[self._HeapSize],self._array[1]
    self._array[self._HeapSize]=None
    self._HeapSize-=1
    self.sink()
    return x

  def sort(self):
    arr=np.array([None]*(self._HeapSize))
    for i in range (self._HeapSize):
      arr[i]=self.extractMin()
    for i in arr:
      self.insert(i)
    return arr

  def print(self):
    if len( self._array)> self._HeapSize:
      for i in range(1,self._HeapSize+1):
        print(self._array[i],end=" ")
    else:
      for i in range(1,len( self._array)):
        print(self._array[i],end=" ")
