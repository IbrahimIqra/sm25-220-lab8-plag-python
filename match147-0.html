class MinHeap:
    def __init__(self, c):
        if c <= 0:
            self.c = 1
        else:
            self.c = c
        self.a = [0] * self.c
        self.n = c
    
    def swim(self, i):
        while i > 0:


            p = (i - 1) // 2
            if self.a[i] < self.a[p]:
                self.a[i], self.a[p] = self.a[p], self.a[i]
                i = p
            else:
                break
    
    def add(self, v):
        if self.n >= self.c:
            return
        self.a[self.n] = v
        j = self.n
        self.n += 1
        self.swim(j)
    
    def sink(self, i):
        for _ in range(self.n):
            if 2 * i + 1 >= self.n:
                break


            l = 2 * i + 1
            r = 2 * i + 2
            s = i
            if l < self.n and self.a[l] < self.a[s]:
                s = l
            if r < self.n and self.a[r] < self.a[s]:
                s = r
            if s != i:
                self.a[i], self.a[s] = self.a[s], self.a[i]
                i = s
            else:
                break
        
    def getMin(self):
        if self.n == 0:
            return None
        m = self.a[0]
        if self.n > 1:
            self.a[0] = self.a[self.n - 1]
        self.n -= 1
        if self.n > 0:
            self.sink(0)
        return m

def solve(t, m):
    h = MinHeap(m)
    i = 0
    while i < len(t):
        x = h.getMin()
        y = x + t[i]
        h.add(y)
        i += 1
    return h.a[:h.n]

if __name__ == "__main__":
    t = [2, 4, 7, 1, 6]
    m = 4
    r = solve(t, m)
    print("Machine loads:", r)

class MaxHeap:
    def __init__(self, c):
        if c <= 0:
            self.__c = 1
        else:
            self.__c = c
        self.__h = [0] * self.__c
        self.__s = 0
    def insert(self, v):
        if self.__s >= self.__c:
            return
        self.__h[self.__s] = v
        i = self.__s
        self.__s += 1
        self.__swim(i)
    def __swim(self, i):
        for _ in range(self.__s):
            if i <= 0:
                break


            p = (i - 1) // 2
            if self.__h[i] > self.__h[p]:
                self.__h[i], self.__h[p] = self.__h[p], self.__h[i]
                i = p
            else:
                break
    def extractMax(self):
        if self.__s == 0:
            return None
        m = self.__h[0]
        if self.__s > 1:
            self.__h[0] = self.__h[self.__s - 1]
        self.__s -= 1
        if self.__s > 0:
            self.__sink(0)
        return m
    def __sink(self, i):
        for _ in range(self.__s):
            if 2 * i + 1 >= self.__s:
                break


            l = 2 * i + 1
            r = 2 * i + 2
            lg = i
            if l < self.__s and self.__h[l] > self.__h[lg]:
                lg = l
            if r < self.__s and self.__h[r] > self.__h[lg]:
                lg = r
            if lg != i:
                self.__h[i], self.__h[lg] = self.__h[lg], self.__h[i]
                i = lg
            else:
                break
def find_top_k_largest(nums, k):
    n = len(nums)
    h = MaxHeap(n)
    for num in nums:
        h.insert(num)
    r = [0] * k
    i = 0
    while i < k:
        r[i] = h.extractMax()
        i += 1
    return r

    k = 3
    o = find_top_k_largest(nums, k)
    print("Top k largest elements:", o)
