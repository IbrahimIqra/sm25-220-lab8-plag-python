class MinHeap:
    def __init__(self, size):
        self.heap = [0] * size
        self.size = size
        self.capacity = size

    def build(self):
        for i in range((self.size - 1) // 2, -1, -1):
            self.down(i)

    def up(self, i):
        if i == 0:
            return



        parent = (i - 1) // 2

        if self.heap[i] < self.heap[parent]:
            temp = self.heap[i]
            self.heap[i] = self.heap[parent]
            self.heap[parent] = temp
            self.up(parent)

    def down(self, i):
        smallest = i

        left = (2 * i) + 1
        right = (2 * i) + 2



        if left < self.size and self.heap[left] < self.heap[smallest]:
            smallest = left

        if right < self.size and self.heap[right] < self.heap[smallest]:
            smallest = right

        if smallest != i:
            temp = self.heap[i]

            self.heap[i] = self.heap[smallest]
            self.heap[smallest] = temp

            self.down(smallest)

    def extract_minimum(self):
        if self.size == 0:
            return None



        minimum = self.heap[0]

        self.heap[0] = self.heap[self.size - 1]
        self.size -= 1
        self.down(0)

        return minimum

    def insert(self, value):


        self.heap[self.size] = value
        self.size += 1
        self.up(self.size - 1)

    def array(self):
        result = [0] * self.size
        for i in range(self.size):
            result[i] = self.heap[i]
        return result


def distribute_tasks(tasks, m):
    heap = MinHeap(m)

    for i in range(m):
        heap.heap[i] = 0

    heap.build()

    for task in tasks:
        minimum = heap.extract_minimum()
        load = minimum + task

        heap.insert(load)

    return heap.array()


tasks = [2, 4, 7, 1, 6]
m = 4
result = distribute_tasks(tasks, m)
print(result)

class MaxHeap:
    def __init__(self, size):
        self.heap = [0] * size
        self.size = size

    def build(self, array):
        for i in range(self.size):
            self.heap[i] = array[i]

        for i in range(self.size // 2 - 1, -1, -1):
            self.down(i)

    def down(self, i):
        largest = i

        left = (2 * i) + 1
        right = (2 * i) + 2



        if left < self.size and self.heap[left] > self.heap[largest]:
            largest = left

        if right < self.size and self.heap[right] > self.heap[largest]:
            largest = right

        if largest != i:
            temp = self.heap[i]
            self.heap[i] = self.heap[largest]
            self.heap[largest] = temp
            self.down(largest)

    def extract_maximum(self):
        if self.size == 0:
            return None



        maximum = self.heap[0]

        self.heap[0] = self.heap[self.size - 1]
        self.size -= 1
        self.down(0)

        return maximum


def largest(nums, k):
    heap = MaxHeap(len(nums))
    heap.build(nums)



    result = [0] * k
    for i in range(k):
        result[i] = heap.extract_maximum()

    return result


nums = [4, 10, 2, 8, 6, 7]
k = 3
result = largest(nums, k)
print(result)
