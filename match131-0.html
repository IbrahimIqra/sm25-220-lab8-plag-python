from array import array
def mk(n, v=0):
    return array('i', (v for _ in range(n)))
def fmt(a):
    s = "["
    i = 0
    m = len(a)
    while i < m:
        s += str(a[i])
        if i + 1 < m:
            s += ", "
        i += 1
    s += "]"
    return s
class Hmin:
    def __init__(self, c):
        self.__a = mk(c, 0)
        self.__n = 0


        self.__c = c
    def _p(self, i): return (i - 1) // 2
    def _l(self, i): return 2 * i + 1
    def _r(self, i): return 2 * i + 2
    def _sw(self, i, j):
        t = self.__a[i]
        self.__a[i] = self.__a[j]
        self.__a[j] = t
    def insert(self, x):
        if self.__n >= self.__c:
            raise IndexError("full")
        i = self.__n
        self.__a[i] = x
        self.__n += 1
        self.swim(i)
    def swim(self, i):


        while i > 0:
            p = self._p(i)
            if self.__a[p] <= self.__a[i]:
                break
            self._sw(i, p)
            i = p
    def extractMin(self):
        if self.__n == 0:
            raise IndexError("empty")
        x = self.__a[0]
        self.__n -= 1
        self.__a[0] = self.__a[self.__n]
        self.sink(0)
        return x

        while True:
            l = self._l(i)
            if l >= self.__n:
                break

            if r < self.__n:

                    ch = r
                else:
                    ch = l
            else:
                ch = l
            if self.__a[i] <= self.__a[ch]:
                break
            self._sw(i, ch)
            i = ch

        out = mk(self.__n, 0)
        i = 0

            out[i] = self.__a[i]
