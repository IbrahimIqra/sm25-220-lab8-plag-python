class MinHeap:
  def __init__(self, capacity):
    self.heap =[0] *capacity # heap array shuru te shob 0
    self.size =capacity
  # parent, left, right index ber korar jonno


  def parent(self,i): 
    return (i-1)//2
  def left(self,i): 
    return 2*i+1
  def right(self, i): 
    return 2*i+2
  def swap(self, i, j):
    self.heap[i], self.heap[j] = self.heap[j], self.heap[i]
  def extractMin(self):
    if self.size == 0:
      return None


    root = self.heap[0]
    self.heap[0] = self.heap[self.size-1]
    self.size -= 1
    self.sink(0)
    return root
  def insert (self, val): # notun value insert kora
    self.size +=1
    i = self.size-1
    self.heap[i ]= val
    self.swim(i )
  def swim(self,i): # upor dike uthano (swim)
    while i>0 and  self.heap[self.parent(i )]>self.heap[i]:
      self.swap(i,self.parent(i))
      i =self.parent(i)
  # niche namano (sink)
  def sink(self, i):
    smallest = i


    l,r = self.left(i), self.right(i)
    if l < self.size and self.heap[l] < self.heap[smallest]:
        smallest = l
    if r< self.size  and self.heap[r] <self.heap[smallest]:
      smallest = r
    if  smallest != i:
      self.swap(i, smallest)
      self.sink(smallest)
# Task 3 function
def task3(tasks, m):
  heap = MinHeap(m)
  for task in tasks:
    minLoad = heap.extractMin()
    updated = minLoad + task
    heap.insert(updated)
  return sorted(heap.heap[:heap.size]) #otherwise 2 7 47 asche

class MaxHeap:

    self.heap = [0]*capacity
    self.size = 0


  def parent(self, i): 
    return (i-1)//2
  def left(self, i): 
    return 2*i+1
  def right(self, i): 
    return 2*i+2
  def swap(self,i,j):
    self.heap[i],self.heap[j] =self.heap[j], self.heap[i]
  def insert(self,  val):


    self.heap[self.size] = val
    self.size += 1
    self.swim(self.size-1)
  def extractMax( self):
    if  self.size == 0: return None


    root= self.heap[0]
    self.heap[0] =self.heap[self.size-1]
    self.size -=1
    self.sink( 0)
    return root
  def  swim(self, i):
    while i> 0 and self.heap[self.parent(i )] < self.heap[ i]:
      self.swap(i,self.parent(i))
      i = self.parent(i)
  def sink(self, i):
    largest = i


    l, r = self.left(i), self.right(i)
    if l  < self.size and   self.heap[l]  > self.heap[largest]:
      largest = l
    if r < self.size  and  self.heap[r]>  self.heap[largest]:
      largest= r
    if largest !=  i:
      self.swap(i , largest)
      self.sink(largest)

def large( nums, k):
  heap =  MaxHeap(len (nums))
  for   num in nums:
    heap.insert(num)
  result = []
  for _ in range(k):
    result.append(heap.extractMax())
  return result
