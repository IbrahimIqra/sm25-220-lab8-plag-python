    self.__cap=cap
    self.__size=0
    self.__heap=[0]*cap

  def getHeap(self):
    return self.__heap

  def insert(self,value):


    if self.__size==self.__cap:
      return
    self.__heap[self.__size]=value
    self.__size+=1
    self.swim(self.__size - 1)

  def extractMin(self):


    if self.__size==0:
      return None
    root=self.__heap[0]
    self.__heap[0]=self.__heap[self.__size - 1]
    self.__heap[self.__size - 1]=0
    self.__size-=1
    self.sink(0)
    return root

      i=(i - 1)//2

        j+=1

        break
      self.__heap[i],self.__heap[j]=self.__heap[j],self.__heap[i]
      i=j

class MaxHeap:
  def __init__(self,cap):
    self.__cap=cap
    self.__size=0
    self.__heap=[0]*cap

  def getHeap(self):
    return self.__heap

  def insert(self,value):


    if self.__size==self.__cap:
      return
    self.__heap[self.__size]=value
    self.__size += 1
    self.swim(self.__size - 1)

  def extractMax(self):


    if self.__size==0:
      return None
    root=self.__heap[0]
    self.__heap[0]=self.__heap[self.__size - 1]
    self.__heap[self.__size - 1]=0
    self.__size-=1
    self.sink(0)
    return root

  def swim(self,i):
    while i>0 and self.__heap[(i - 1)//2]<self.__heap[i]:
      self.__heap[(i - 1)//2],self.__heap[i]=self.__heap[i],self.__heap[(i - 1)//2]
      i=(i - 1)//2



  def sink(self,i):
    while 2*i+1<self.__size:
      j=2*i+1
      if j+1<self.__size and self.__heap[j+1]>self.__heap[j]:
        j+=1
      if self.__heap[i]>=self.__heap[j]:
        break
      self.__heap[i],self.__heap[j]=self.__heap[j],self.__heap[i]
      i=j


def top_k_largest(nums,k):
  n=len(nums)
  heap=MaxHeap(n)
  for j in range(n):
    heap.insert(nums[j])
  result=[0]*k
  for j in range(k):
    result[j]=heap.extractMax()
  return result
