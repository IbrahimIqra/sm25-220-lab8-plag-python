class MinHeap:
  def __init__(self, task, m):
      self.task = task
      self.machines = [0] * m


      self.size = m




  def swim(self,index):
      while index > 0:
          parent = (index-1) // 2
          if self.machines[parent] > self.machines[index]:
              self.machines[parent],self.machines[index] = self.machines[index],self.machines[parent]
              index = parent
          else:
              break




  def sink(self,index):
      while True:
          left = 2 * index + 1
          right = 2 * index + 2
          smallest = index




          if left < self.size and self.machines[left] <= self.machines[smallest]:
              smallest = left


          if right < self.size and self.machines[right] <= self.machines[smallest]:
              smallest = right




          if smallest != index:
              self.machines[index], self.machines[smallest] = self.machines[smallest], self.machines[index]
              index = smallest
          else:
              break




  def insert(self,value):


      self.machines[self.size] = value
      self.swim(self.size)
      self.size+=1




  def extract(self):
      if self.size == 0 :
          print('Heap is Empty')
          return None




      min_value = self.machines[0]
      self.machines[0] = self.machines[self.size-1]
      self.machines[self.size-1] = None
      self.size -= 1
      self.sink(0)
      return min_value




  def main_function(self):
      for i in range(len(self.task)):
          x = self.extract()
          value = self.task[i]+x
          self.insert(value)
      return self.machines








tasks = [2, 4, 7, 1, 6]
print('Total Tasks :',tasks)
m = 4
print('Total Machines',m)
heap = MinHeap(tasks, m)
x = heap.main_function()
print("Final machine loads:", x)

class MaxHeap:
  def __init__(self,nums,k):
      self.heap=nums
      self.result=[None]*k
      self.size=len(nums)
      self.max_heap()




  def max_heap(self):
      for i in range(self.size//2,-1,-1):
          self.sink(i)




  def sink (self,index):
      while True:
          largest=index
          l = 2*index + 1
          r = 2*index + 2
          if l < self.size and self.heap[l] >= self.heap[largest]:
              largest = l


          if r < self.size and self.heap[r] >= self.heap[largest]:
              largest = r
          if largest != index :
              self.heap[largest],self.heap[index] = self.heap[index],self.heap[largest]
              index = largest
          else:
              break
  def swim(self,index):
      while index > 0:
          parent = (index-1)//2


          if self.heap[parent] < self.heap[index]:
              self.heap[parent],self.heap[index] = self.heap[index],self.heap[parent]
              index = parent
          else:
              break
  def extract(self):


      removed=self.heap[0]
      self.heap[0]=self.heap[self.size-1]
      self.heap[self.size-1]=None
      self.size-=1
      self.sink(0)
      return removed




  def main_function(self):
      for i in range(len(self.result)):
          x = self.extract()
          self.result[i] = x
      return self.result
