class MaxHeap:
    def __init__(self, capacity):
        self.__heap = np.zeros(capacity+1, dtype=int) 
        self.__size = 0
        self.__capacity = capacity

    def insert(self, value):
        if self.__size >= self.__capacity:
            raise Exception("Heap is full")


        self.__size += 1
        self.__heap[self.__size] = value
        self.swim(self.__size)

    def swim(self, index):
        if index <= 1:
            return
        parent = index // 2
        if self.__heap[parent] < self.__heap[index]: 


            self.__heap[parent], self.__heap[index] = self.__heap[index], self.__heap[parent]
            self.swim(parent)

    def deleteMax(self):
        if self.__size == 0:
            raise Exception("Heap is empty")


        max_value = self.__heap[1]
        self.__heap[1] = self.__heap[self.__size]
        self.__size -= 1
        self.sink(1)
        return max_value

    def sink(self, index):
        left = 2 * index
        right = 2 * index + 1


        largest = index

        if left <= self.__size and self.__heap[left] > self.__heap[largest]:
            largest = left
        if right <= self.__size and self.__heap[right] > self.__heap[largest]:
            largest = right

        if largest != index:
            self.__heap[largest], self.__heap[index] = self.__heap[index], self.__heap[largest]
            self.sink(largest)

    def extractMax(self):
        return self.deleteMax()

    def heapSort(self):
        result = []
        while self.__size > 0:
            result.append(self.extractMax())
        return result

def distribute_tasks(tasks, m):
    heap=MinHeap(m)
    for i in range(m):
        heap.insert(0)
    for task in tasks:
        min_machine = heap.extractMin()
        heap.insert(min_machine + task)

    return heap.Min_sort()

def array_to_maxHeap(arr):
    Heap=MaxHeap(len(arr))
    for num in arr:
        Heap.insert(num)
    return Heap
def k_largest(arr, k):
    max_heap = array_to_maxHeap(arr)
    result = []
    for _ in range(k):
        result.append(max_heap.extractMax())
    return result
