class MinHeap:
  def __init__(self, capacity):
    self._heap = [0] * capacity
    self._capacity = capacity
    self._size = 0

  def insert(self, value):
    if self._size == len(self._heap):
      return
    self._heap[self._size] = value
    self._swim(self._size)
    self._size += 1

  def _swim(self, index):
    if index == 0:
      return


    store = (index - 1) // 2
    if self._heap[index] < self._heap[store]:
      self._heap[index], self._heap[store] = self._heap[store], self._heap[index]
      self._swim(store)

  def extractMin(self):
    if self._size == 0:
      return None
    low_value = self._heap[0]
    self._size -= 1
    self._heap[0] = self._heap[self._size]
    self._sink(0)
    return low_value

  def _sink(self, index):


    left = 2 * index + 1
    right = 2 * index + 2
    lowest = index
    if left < self._size and self._heap[left] < self._heap[lowest]:
      lowest = left
    if right < self._size and self._heap[right] < self._heap[lowest]:
      lowest = right
    if lowest != index:
      self._heap[index], self._heap[lowest] = self._heap[lowest], self._heap[index]
      self._sink(lowest)

  def sort(self):
    real_size = self._size
    for i in range(self._size - 1, 0, -1):
      self._heap[0], self._heap[i] = self._heap[i], self._heap[0]
      self._size -= 1
      self._sink(0)
    self._size = real_size

  def printHeap(self):
    return self._heap[:self._size]

  def checkEmpty(self):
    return self._size == 0

  def checkSize(self):
    return self._size

class MaxHeap:
  def __init__(self, capacity):
    self.__capacity = capacity
    self.__heap = [0] * capacity  # Array representing the heap
    self.__size = 0               # Current size of the heap

  def insert(self, value):
    if self.__size == self.__capacity:
      return False
    self.__heap[self.__size] = value
    self.__size += 1
    self.__swim(self.__size - 1)
    return True

  def __swim(self, index):
    if index > 0:
      return


    store = (index - 1) // 2
    if self.__heap[index] > self.__heap[store]:
      self.__heap[index], self.__heap[store] = self.__heap[store], self.__heap[index]

  def extractMax(self):
    if self.__size <= 0:
      return None
    highest_value = self.__heap[0]
    self.__heap[0] = self.__heap[self.__size - 1]
    self.__size -= 1
    self.__sink(0)
    return highest_value

  def __sink(self, index):


    left = 2 * index + 1
    right = 2 * index + 2
    biggest = index

    if left < self.__size and self.__heap[left] > self.__heap[biggest]:
      biggest = left

    if right < self.__size and self.__heap[right] > self.__heap[biggest]:
      biggest = right

    if biggest != index:
      self.__heap[index], self.__heap[biggest] = self.__heap[biggest], self.__heap[index]
      self.__sink(biggest)

  def sort(self):
      real_size = self.__size
      for i in range(self.__size - 1, 0, -1):
        self.__heap[0], self.__heap[i] = self.__heap[i], self.__heap[0]
        self.__size -= 1
        self.__sink(0)
      sort_arr = self.__heap[:real_size]
      self.__size = real_size  # Restore the original heap size
      return sort_arr

  def printHeap(self):
    return self.__heap[:self.__size]

  def checkSize(self):
    return self.__size

  def checkEmpty(self):
    return self.__size == 0

# Example usage
heap = MaxHeap(10)

heap.insert(1)
heap.insert(6)
heap.insert(5)
heap.insert(2)
heap.insert(4)
print("A Heap:", heap.printHeap())
print("After sorting:", heap.sort())
