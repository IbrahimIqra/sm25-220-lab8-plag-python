    self.swim(self.__size -1)

    if self.__heap[idx] <self.__heap[parent_node]:

          s=left

   self.__heap[self.__size]= val

   self.swim(self.__size -1)
  def swim(self,i):
   while i>0:
    parent_node= (i-1)//2
    if self.__heap[i]>self.__heap[parent_node]:

        large_val= right
