class MinHeap:
    def __init__(self, capacity):
        self.capacity = capacity
        self.size = 0
        self.heap = [0] * capacity



    def insert(self, value):
        if self.size == self.capacity:
            return
        self.heap[self.size] = value
        self.size += 1
        self._swim(self.size - 1)

    def extractMin(self):
        if self.size == 0:
            return None
        root = self.heap[0]
        self.heap[0] = self.heap[self.size - 1]
        self.size -= 1
        self._sink(0)
        return root

    def _swim(self, i):
        while i > 0:
            parent = (i - 1) // 2
            if self.heap[i] < self.heap[parent]:
                self.heap[i], self.heap[parent] = self.heap[parent], self.heap[i]
                i = parent
            else:
                break

    def _sink(self, i):
        while 2 * i + 1 < self.size:
            left = 2 * i + 1
            right = 2 * i + 2
            smallest = left
            if right < self.size and self.heap[right] < self.heap[left]:
                smallest = right


            if self.heap[i] <= self.heap[smallest]:
                break
            self.heap[i], self.heap[smallest] = self.heap[smallest], self.heap[i]
            i = smallest

    def getArray(self):
        return self.heap[:self.size]




def assignTasks(tasks, m):
    heap = MinHeap(m)
    for _ in range(m):
        heap.insert(0)

    for t in tasks:
        load = heap.extractMin()
        newLoad = load + t
        heap.insert(newLoad)
    result = heap.getArray()
    result.sort()
    return result

tasks = [2, 4, 7, 1, 6]
m = 4
result = assignTasks(tasks, m)
print("Final Heap Array:", result)

class MaxHeap:
    def __init__(self, capacity):
        self.capacity = capacity
        self.size = 0
        self.heap = [0] * capacity



    def insert(self, value):
        if self.size == self.capacity:
            return
        self.heap[self.size] = value
        self.size += 1
        self._swim(self.size - 1)

    def extractMax(self):
        if self.size == 0:
            return None
        maxVal = self.heap[0]
        self.heap[0] = self.heap[self.size - 1]
        self.size -= 1
        self._sink(0)
        return maxVal

    def _swim(self, idx):
        while idx > 0:
            parent = (idx - 1) // 2
            if self.heap[idx] > self.heap[parent]:
                self.heap[idx], self.heap[parent] = self.heap[parent], self.heap[idx]
                idx = parent
            else:
                break

    def _sink(self, idx):
        while 2 * idx + 1 < self.size:
            left = 2 * idx + 1
            right = 2 * idx + 2
            bigger = left
            if right < self.size and self.heap[right] > self.heap[left]:
                bigger = right


            if self.heap[idx] >= self.heap[bigger]:
                break
            self.heap[idx], self.heap[bigger] = self.heap[bigger], self.heap[idx]
            idx = bigger

def topKlargest(nums, k):
    n = len(nums)
    heap = MaxHeap(n)

    for val in nums:


        heap.insert(val)

    result = [0] * k
    for i in range(k):
        result[i] = heap.extractMax()

    return result
