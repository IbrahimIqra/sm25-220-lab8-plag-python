class MinHeap:
    def __init__(self,size):
        self.__limit=size
        self.__heap=[0]*size
        self.__count=0


    def Insert(self,val):
        if self.__count==self.__limit:
            return
        self.__heap[self.__count]=val
        self.__count+=1
        self.ShiftUp(self.__count-1)


    def ShiftUp(self,idx):
        while idx>0:
            parent=(idx-1)//2
            if self.__heap[idx]<self.__heap[parent]:
                t=self.__heap[idx]
                self.__heap[idx]=self.__heap[parent]
                self.__heap[parent]=t
                idx=parent
            else:
                break
    def extractMin(self):
        if self.__count==0:
            return None


        m=self.__heap[0]
        self.__heap[0]=self.__heap[self.__count-1]
        self.__count-=1
        self.ShiftDown(0)
        return m
    def ShiftDown(self,idx):
        while (2*idx+1)<self.__count:
            left=2*idx+1


            right=left+1
            s=left
            if right<self.__count and self.__heap[right]<self.__heap[left]:
                s=right


            if self.__heap[s]<self.__heap[idx]:
                t=self.__heap[idx]
                self.__heap[idx]=self.__heap[s]
                self.__heap[s]=t
                idx=s
            else:
                break

def assign_tasks(tasks,m):
    if m<=0:
        return []
    h=MinHeap(m)
    i=0
    while i<m:
        h.Insert(0)
        i+=1
    j=0
    n=len(tasks)
    while j<n:
        x=h.extractMin()
        if x is None:
            x=0
        x=x+tasks[j]
        h.Insert(x)
        j+=1
    res=[0]*m
    k=0
    while k<m:
        t=h.extractMin()
        if t is None:
            t=0
        res[k]=t
        k+=1
    return res

    def ShiftDown(self,idx):

            if right<self.__count and self.__heap[right]>self.__heap[left]:

                self.__heap[idx]=self.__heap[l]

def top_k_largest(nums,k):
    n=len(nums)
    h=MaxHeap(n)
    i=0
    while i<n:
        h.Insert(nums[i])
        i+=1
    res=[0]*k
    j=0
    while j<k:
        res[j]=h.extractMax()
        j+=1
    return resS
